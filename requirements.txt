# Raspberry Pi 4B setup notes for Arclight
#
# Recommended system packages:
#   sudo apt update && sudo apt upgrade -y
#   sudo apt install -y \
#       python3-pip python3-venv python3-dev \
#       build-essential cmake git \
#       libatlas-base-dev libopenblas-dev \
#       libjpeg-dev libpng-dev libtiff-dev \
#       libavcodec-dev libavformat-dev libswscale-dev \
#       libgtk-3-dev \
#       python3-opencv python3-rpi.gpio
#
# Recommended virtualenv:
#   python3 -m venv arclight_env --system-site-packages
#   source arclight_env/bin/activate
#   pip install -r requirements.txt
#
# Notes:
# - face_recognition depends on dlib and can take time to install on a Pi 4B.
# - HOG detection is used by default for CPU performance.
# - The large encoding model is used by default because the copied encodings were trained that way.
# - tflite-runtime is preferred over full TensorFlow for the anti-spoof model.
