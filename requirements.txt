# Raspberry Pi OS setup for Arclight

# Step 1: Update system packages
sudo apt update && sudo apt upgrade -y

# Step 2: Install system dependencies
sudo apt install -y \
    python3-pip python3-venv \
    python3-opencv python3-rpi.gpio \
    build-essential cmake git pkg-config \
    libatlas-base-dev libopenblas-dev \
    libjpeg-dev libpng-dev libtiff-dev \
    libavcodec-dev libavformat-dev libswscale-dev \
    libgtk-3-dev

# Step 3: Run the remaining commands from the project folder.

# Step 4: Create virtual environment.
# --system-site-packages lets the venv use apt-installed Pi packages such as
# cv2 and RPi.GPIO.
python3 -m venv --system-site-packages venv

# Step 5: Activate virtual environment
source venv/bin/activate

# Step 6: Create Python package requirements for pip-installed app packages
cat << EOF > requirements_pi.txt
fastapi
uvicorn[standard]
python-multipart
numpy
pandas
openpyxl
insightface
onnxruntime
pillow
tflite-runtime
EOF

# Step 7: Install Python packages
pip install --upgrade pip
pip install -r requirements_pi.txt

# Step 8: Check if everything installed successfully
python -c "import cv2, numpy, fastapi, insightface, RPi.GPIO, tensorflow as tf; assert tf.lite.Interpreter; print('Core packages installed successfully')"
